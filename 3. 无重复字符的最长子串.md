# 3. 无重复字符的最长子串

###### 给定一个字符串 `s` ，请你找出其中不含有重复字符的 **最长子串** 的长度。



一开始没有思路，后来看了题解才知道。

这题需要关注到字符**子**串的**连续性**，利用这个性质借助队列的想法就比较容易求解。

```C++
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        string maxSubStr = "";
        int maxLength = 0;
        for(int i=0;i<s.length();i++)
        {
            char t = s[i];
            int index = maxSubStr.find(t);
            if(index == maxSubStr.npos)
                maxSubStr += t;
            else
            {
                int l = maxSubStr.length();
                if(l > maxLength)
                    maxLength = l;
                maxSubStr = maxSubStr.substr(index, l);//从index开始往后l个长度，多于字符串长度则截取到endl
                //error就是这步，index错了
            }
        }
        // cout<<maxSubStr;
        return maxLength;
    }
};
```

"aab"

"aabaab!bb"

```C++
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        string maxSubStr = "";
        int maxLength = 0;
        for(int i=0;i<s.length();i++)
        {
            char t = s[i];
            int index = maxSubStr.find(t);
            if(index == maxSubStr.npos)
            {
                maxSubStr += t;
                // cout<<1<<maxSubStr<<endl;
            }
            else
            {
                int l = maxSubStr.length();
                if(l > maxLength)
                    maxLength = l;
                maxSubStr = maxSubStr.substr(index+1, l);//从index开始往后l个长度，多于字符串长度则截取到endl
                maxSubStr += t;
                // cout<<index <<maxSubStr<<endl;
            }
        }
        int l = maxSubStr.length();
        if(l > maxLength)
            maxLength = l;
        return maxLength;
    }
};
```



# 其他解法

非常强，且答主对“滑动窗口”进行了归纳

因为只需要返回长度，所以可以使用**集合**作为容器

```C++
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        if(s.size() == 0) return 0;
        unordered_set<char> lookup;
        int maxStr = 0;
        int left = 0;
        for(int i = 0; i < s.size(); i++){
            while (lookup.find(s[i]) != lookup.end()){
                lookup.erase(s[left]);
                left ++;
            }
            maxStr = max(maxStr,i-left+1);
            lookup.insert(s[i]);
    }
        return maxStr;
        
    }
};
```
